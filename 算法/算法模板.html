# 最大公约数和最小公倍数

// 计算最小公倍数：a / gcd(a,b) * b
private int lcm(int a, int b){
    return a/gcd(a,b) * b;
}

// 计算最大公约数（欧几里得算法）
private int gcd(int a,int b){
    while(b != 0){
        int t = a % b;
        a = b;
        b = t;
    }
    return a;
}
